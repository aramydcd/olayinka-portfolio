import os
import dj_database_url # Import the utility you just installed
from pathlib import Path

# Build paths inside the project like this: BASE_DIR / 'subdir'.
BASE_DIR = Path(__file__).resolve().parent.parent

# --- DATABASE CONFIGURATION ---
# 1. Get the URL from Render's environment variables
DATABASE_URL = os.getenv('DATABASE_URL')

if DATABASE_URL:
    # 2. Use dj-database-url to automatically configure Postgres
    # It handles the "postgres://" vs "postgresql://" fix for you!
    DATABASES = {
        'default': dj_database_url.config(
            default=DATABASE_URL,
            conn_max_age=600, # Keeps connections alive for better performance
            conn_health_checks=True,
        )
    }
    print("✅ Django is using the Remote PostgreSQL Database")
else:
    # 3. Fallback to local SQLite for your local development
    DATABASES = {
        'default': {
            'ENGINE': 'django.db.backends.sqlite3',
            'NAME': BASE_DIR / 'db.sqlite3',
        }
    }
    print("⚠️ DATABASE_URL not found, Django is using local SQLite")
